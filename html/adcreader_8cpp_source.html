183 uint16_t Adafruit_ADS1015::readADC_SingleEnded(uint8_t channel) {
184  if (channel > 3)
185  {
186  return 0;
187  }
188 
189  // Start with default values
190  uint16_t config = ADS1015_REG_CONFIG_CQUE_NONE | // Disable the comparator (default val)
191  ADS1015_REG_CONFIG_CLAT_NONLAT | // Non-latching (default val)
192  ADS1015_REG_CONFIG_CPOL_ACTVLOW | // Alert/Rdy active low (default val)
193  ADS1015_REG_CONFIG_CMODE_TRAD | // Traditional comparator (default val)
194  ADS1015_REG_CONFIG_DR_1600SPS | // 1600 samples per second (default)
195  ADS1015_REG_CONFIG_MODE_SINGLE; // Single-shot mode (default)
196 
197  // Set PGA/voltage range
198  config |= m_gain;
199 
200  // Set single-ended input channel
201  switch (channel)
202  {
203  case (0):
204  config |= ADS1015_REG_CONFIG_MUX_SINGLE_0;
205  break;
206  case (1):
207  config |= ADS1015_REG_CONFIG_MUX_SINGLE_1;
208  break;
209  case (2):
210  config |= ADS1015_REG_CONFIG_MUX_SINGLE_2;
211  break;
212  case (3):
213  config |= ADS1015_REG_CONFIG_MUX_SINGLE_3;
214  break;
215  }
216 
217  // Set 'start single-conversion' bit
218  config |= ADS1015_REG_CONFIG_OS_SINGLE;
219 
220  // Write config register to the ADC
221  writeRegister(m_i2cFd, m_i2cAddress, ADS1015_REG_POINTER_CONFIG, config);
222 
223  // Wait for the conversion to complete
224  usleep(1000*m_conversionDelay);
225 
226  // Read the conversion results
227  // Shift 12-bit results right 4 bits for the ADS1015
228  return readRegister(m_i2cFd, m_i2cAddress, ADS1015_REG_POINTER_CONVERT) >> m_bitShift;
229 }

245  ADS1015_REG_CONFIG_DR_1600SPS | // 1600 samples per second (default)

247 
248  // Set PGA/voltage range

262
